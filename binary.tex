#include <stdio.h>
int binarySearch(int a[], int beg, int end, int search)
{
    int mid;
    if(end >= beg)
    {        mid = (beg + end)/2;
        if(a[mid] == search)
        {
            return mid+1;
        }

        else if(a[mid] < search)
        {
            return binarySearch(a, mid+1, end, search);
        }
        else
        {
            return binarySearch(a, beg, mid-1, search);
        }
    }
    return -1;
}
